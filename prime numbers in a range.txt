#include<stdio.h>
int main()
{
   int i,a=1,b=100,count,n;
   for(n=a;n<=b;n++)
   {
     count=0;
       for(i=2;i<=n/2;i++)
       {
           if(n%i==0)
           {
           count++;
           break;
           }
       }
    
   if(count==0 && n>1 )
   {
   printf("% d",n);
   }
   }
}
