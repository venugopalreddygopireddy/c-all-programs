#include <stdio.h>
#include<string.h>
int main() 
{
 char string[20]="madam";
 int i,flag=0,length;
 length=strlen(string);
 for(i=0;i<length/2;i++)
 {
     if(string[i]!=string[length-i-1])
     {
        flag=1;
        break;
     }
 }
 if(flag==1)
 {
 printf("%s is   not a palindrome",string);
 }
 else
 {
 printf("%s is  a palindrome",string);
 }
  return 0;
}
